class Checkout {
    private ArrayList<Integer> array;

    public Checkout() {
    this.array = new ArrayList<>();
    }
    
    public int get_max() {
    int max = -1;
    for(int i =0;i<array.size();i++){
    if(max<array.get(i)) max = array.get(i);
    }
    return max;
    }
    
    public void add(int value) {
    array.add(value);
    }
    
    public int remove() {
    if(array.size() == 0) return -1;
    return array.remove(0);
    }
}
